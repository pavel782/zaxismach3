Dim closeResult, mach3CommonVariablesStartIndex, spindleSwitch, enablePluginmach3CommonVariablesStartIndex = {0}spindleSwitch = GetVar(mach3CommonVariablesStartIndex + 6)enablePlugin = GetVar(mach3CommonVariablesStartIndex + 7)if (spindleSwitch = 1 or enablePlugin <> 1) then    SetVar(mach3CommonVariablesStartIndex + 6, 0)  DoSpinStop()else  Set serialPortAx = CreateObject("SerialPortAxControl.SerialPortAxControl")    closeResult = serialPortAx.CloseSerialPort("cacheKey00", true)  if (closeResult <> "OK") then    MachMsg("Cant close connection with arduino serial port",  "Warning", 0)  end ifend if
